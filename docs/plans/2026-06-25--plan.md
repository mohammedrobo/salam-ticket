  import path from 'path';

  const DB_PATH = path.join(process.cwd(), 'data', 'drivers.db');

  // Ensure data directory exists
  import fs from 'fs';
  const dataDir = path.join(process.cwd(), 'data');
  if (!fs.existsSync(dataDir)) {
    fs.mkdirSync(dataDir, { recursive: true });
  }

  const db = new Database(DB_PATH);

  // Create table if not exists
  db.exec(`
    CREATE TABLE IF NOT EXISTS drivers (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL,
      scanned_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      status TEXT DEFAULT 'waiting'
    )
  `);

  export default db;
  ```

- [ ] **Step 1: Create GET handler**

  import { NextResponse } from 'next/server';
  import db from '@/lib/db';

  export async function GET() {
    const drivers = db.prepare('SELECT * FROM drivers WHERE status = ?').all('waiting');
    return NextResponse.json(drivers);
  }
  ```

- [ ] **Step 2: Create POST handler**
  ```typescript
  export async function POST(request: Request) {
    const { name } = await request.json();
    
    if (!name || name.trim() === '') {
      return NextResponse.json({ error: 'Name is required' }, { status: 400 });
    }

    const result = db.prepare('INSERT INTO drivers (name) VALUES (?)').run(name.trim());
    
    return NextResponse.json({ 
      id: result.lastInsertRowid, 
      name: name.trim(),
      status: 'waiting'
    }, { status: 201 });
  }
  ```

- [ ] **Step 3: Create DELETE handler**
  ```typescript
  export async function DELETE(request: Request) {
    const { searchParams } = new URL(request.url);
    const id = searchParams.get('id');

    if (!id) {
      return NextResponse.json({ error: 'ID is required' }, { status: 400 });
    }

    db.prepare('DELETE FROM drivers WHERE id = ?').run(id);
    
    return NextResponse.json({ success: true });
  }
  ```

## Task 4: Driver Check-in Page

**Files:**
- Create: `app/scan/page.tsx`

- [ ] **Step 1: Create scan page**
  ```tsx
  'use client';

  import { useState, useEffect } from 'react';

  export default function ScanPage() {
    const [name, setName] = useState('');
    const [savedName, setSavedName] = useState('');
    const [submitted, setSubmitted] = useState(false);
    const [error, setError] = useState('');

    useEffect(() => {
      const saved = localStorage.getItem('driverName');
      if (saved) {
        setSavedName(saved);
        setName(saved);
      }
    }, []);

    const handleSubmit = async (e: React.FormEvent) => {
      e.preventDefault();
      setError('');

      if (!name.trim()) {
        setError('Please enter your name');
        return;
      }

      try {
        const res = await fetch('/api/drivers', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ name: name.trim() }),
        });

        if (res.ok) {
          localStorage.setItem('driverName', name.trim());
          setSubmitted(true);
        } else {
          const data = await res.json();
          setError(data.error || 'Something went wrong');
        }
      } catch {
        setError('Network error');
      }
    };

    if (submitted) {
      return (
        <div className="min-h-screen bg-gradient-to-br from-green-900 via-green-800 to-green-900 flex items-center justify-center p-4">
          <div className="bg-white/10 backdrop-blur-lg rounded-3xl p-8 text-center max-w-md w-full">
            <div className="text-6xl mb-4">✓</div>
            <h1 className="text-3xl font-bold text-white mb-2">You're Checked In!</h1>
            <p className="text-green-200 text-lg">{name}</p>
            <p className="text-green-300 mt-4">Go to the manager screen</p>
          </div>
        </div>
      );
    }

    return (
      <div className="min-h-screen bg-gradient-to-br from-blue-900 via-blue-800 to-purple-900 flex items-center justify-center p-4">
        <div className="bg-white/10 backdrop-blur-lg rounded-3xl p-8 max-w-md w-full">
          <h1 className="text-3xl font-bold text-white text-center mb-8">Driver Check-In</h1>
          
          <form onSubmit={handleSubmit} className="space-y-6">
            <div>
              <label className="block text-white mb-2 font-medium">Your Name</label>
              <input
                type="text"
                value={name}
                onChange={(e) => setName(e.target.value)}
                placeholder="Enter your full name"
                className="w-full px-4 py-3 rounded-xl bg-white/10 border border-white/20 text-white placeholder-white/50 focus:outline-none focus:ring-2 focus:ring-blue-400 text-lg"
                autoFocus
              />
            </div>

            {error && (
              <p className="text-red-300 text-center">{error}</p>
            )}

            <button
              type="submit"
              className="w-full py-3 bg-blue-500 hover:bg-blue-600 text-white font-bold rounded-xl transition-colors text-lg"
            >
              Check In
            </button>
          </form>
        </div>
      </div>
    );
  }
  ```

## Task 5: QR Code Generation API

**Files:**
- Create: `app/api/qr/route.ts`

- [ ] **Step 1: Create QR generation route**
  ```typescript
  import { NextResponse } from 'next/server';
  import QRCode from 'qrcode';

  export async function GET(request: Request) {
    const { searchParams } = new URL(request.url);
    const baseUrl = searchParams.get('url') || 'http://localhost:3000/scan';

    const svg = await QRCode.toString(baseUrl, {
      type: 'svg',
      width: 300,
      margin: 2,
      color: {
        dark: '#000000',
        light: '#ffffff',
      },
    });

    return new NextResponse(svg, {
      headers: {
        'Content-Type': 'image/svg+xml',
        'Cache-Control': 'no-cache',
      },
    });
  }
  ```

## Task 6: Manager Dashboard

**Files:**
- Create: `app/page.tsx`

- [ ] **Step 1: Create dashboard page**
  ```tsx
  'use client';

  import { useState, useEffect } from 'react';

  interface Driver {
    id: number;
    name: string;
    scanned_at: string;
    status: string;
  }

  export default function Dashboard() {
    const [drivers, setDrivers] = useState<Driver[]>([]);
    const [loading, setLoading] = useState(true);

    const fetchDrivers = async () => {
      try {
        const res = await fetch('/api/drivers');
        const data = await res.json();
        setDrivers(data);
      } catch (error) {
        console.error('Failed to fetch drivers:', error);
      } finally {
        setLoading(false);
      }
    };

    useEffect(() => {
      fetchDrivers();
      const interval = setInterval(fetchDrivers, 3000);
      return () => clearInterval(interval);
    }, []);

    const handleDelete = async (id: number) => {
      try {
        await fetch(`/api/drivers?id=${id}`, { method: 'DELETE' });
        setDrivers(drivers.filter(d => d.id !== id));
      } catch (error) {
        console.error('Failed to delete driver:', error);
      }
    };

    const formatTime = (dateStr: string) => {
      const date = new Date(dateStr + 'Z');
      return date.toLocaleTimeString('en-US', { 
        hour: '2-digit', 
        minute: '2-digit',
        hour12: true 
      });
    };

    return (
      <div className="min-h-screen bg-gradient-to-br from-slate-900 via-slate-800 to-slate-900">
        <div className="container mx-auto px-4 py-8">
          {/* Header */}
          <div className="flex flex-col lg:flex-row justify-between items-start lg:items-center mb-8 gap-4">
            <div>
              <h1 className="text-4xl font-bold text-white">Driver Dashboard</h1>
              <p className="text-slate-400 mt-1">Manage drivers on-site</p>
            </div>
            
            {/* QR Code Section */}
            <div className="bg-white/10 backdrop-blur-lg rounded-2xl p-4 flex items-center gap-4">
              <img 
                src={`/api/qr?t=${Date.now()}`} 
                alt="Scan to check in"
                className="w-24 h-24"
              />
              <div>
                <p className="text-white font-medium">Scan QR to Check In</p>
                <button 
                  onClick={() => window.location.reload()}
                  className="text-sm text-blue-400 hover:text-blue-300 mt-1"
                >
                  Refresh QR
                </button>
              </div>
            </div>
          </div>

          {/* Driver Count */}
          <div className="bg-white/5 backdrop-blur rounded-xl p-4 mb-6 inline-block">
            <span className="text-slate-400">Waiting Drivers: </span>
            <span className="text-3xl font-bold text-white">{drivers.length}</span>
          </div>

          {/* Driver List */}
          {loading ? (
            <div className="text-center text-slate-400 py-12">Loading...</div>
          ) : drivers.length === 0 ? (
            <div className="text-center py-16">
              <div className="text-6xl mb-4">🚗</div>
              <h2 className="text-2xl font-semibold text-white mb-2">No Drivers Waiting</h2>
              <p className="text-slate-400">Drivers will appear here after scanning the QR code</p>
            </div>
          ) : (
            <div className="grid gap-4">
              {drivers.map((driver) => (
                <div 
                  key={driver.id}
                  className="bg-white/5 backdrop-blur rounded-2xl p-6 flex flex-col sm:flex-row items-start sm:items-center justify-between gap-4 border border-white/10 hover:border-white/20 transition-colors"
                >
                  <div className="flex-1">
                    <h3 className="text-2xl font-bold text-white">{driver.name}</h3>
                    <div className="flex items-center gap-4 mt-2">
                      <span className="text-slate-400">
                        Scanned: {formatTime(driver.scanned_at)}
                      </span>
                      <span className="px-2 py-1 bg-yellow-500/20 text-yellow-300 rounded-full text-sm">
                        {driver.status}
                      </span>
                    </div>
                  </div>
                  
                  <button
                    onClick={() => handleDelete(driver.id)}
                    className="px-6 py-3 bg-red-500 hover:bg-red-600 text-white font-bold rounded-xl transition-colors flex items-center gap-2"
                  >
                    <span>✓</span> Done
                  </button>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    );
  }
  ```
